```python
"""
GhostInjector: pipes holodeck sentiment logs into plato-torch as ghost tiles.
"""
import json
from pathlib import Path
from plato_torch.training_rooms.tile_store import TileStore

class GhostInjector:
    def __init__(self, tile_store: TileStore):
        self.tile_store = tile_store

    def ingest_holodeck_log(self, log_path: Path):
        with open(log_path) as f:
            events = json.load(f)
        for event in events:
            tile = {
                "type": "ghost",
                "source": "holodeck",
                "content": event["dialog"],
                "sentiment": event.get("sentiment", 0.0),
                "lesson": event.get("outcome", "")
            }
            self.tile_store.add(tile)
        print(f"Injected {len(events)} ghost tiles from {log_path}")
```
